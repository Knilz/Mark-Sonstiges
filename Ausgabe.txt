   case 1: var b = "B",break();
   case 2: var b = "C",break();  
   case 3: var b = "D",break();
   case 4: var b = "E",break();
   case 5: var b = "F",break();
   case 6: var b = "G",break();
   case 7: var b = "H",break();
  } 
  position += b;
  position += (achse0+1);
  return position;
}
